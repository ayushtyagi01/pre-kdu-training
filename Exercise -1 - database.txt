create database employee;

CREATE TABLE department (
  department_id INTEGER PRIMARY KEY,
  department_name VARCHAR(30) NOT NULL,
);
CREATE TABLE employee (
  employee_id INTEGER PRIMARY KEY,
  employee_name VARCHAR(30) NOT NULL,
  salary INTEGER,
  department_id,
  FOREIGN KEY (department_id) REFERENCES department(department_id)
);
SELECT employee.employee_id,employee.employee_name,department.department_name
    FROM employee inner join department on department.department_id=employee.department_id;


SELECT employee_name,salary FROM employee order by salary desc;
